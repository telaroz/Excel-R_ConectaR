'Funcion para el kpx
Function kpx(x As Integer, k As Integer, qx As Range, w As Integer)
Dim j As Integer
Dim P As Double
j = 0
P = 1
  If x + k > w Then
    kpx = 0

    Else
    
       For j = 0 To k - 1
       P = P * (1 - qx.Cells(x + j + 1, 1).Value)
       
       Next j
       
    kpx = P
    
End If

End Function

'Anualidad vida entera que paga al inicio del período (período de un año)
Function adospuntos_x(x As Integer, i As Double, qx As Range, w As Integer)
Dim j As Integer
Dim v As Double
Dim r As Double

j = 0
r = 0
v = (1 + i) ^ (-1)
    
For j = 0 To w - 1
r = r + ((v ^ j) * kpx(x, j, qx, w))
Next j

adospuntos_x = r


End Function
'Anualidad vida entera que paga al inicio del período (por m-ésimos)
Function adospuntos_x_m(x As Integer, i As Double, qx As Range, w As Integer)
Dim j As Integer
Dim v As Double
Dim r As Double

j = 0
r = 0
v = (1 + i) ^ (-1)

For j = 0 To w - 1
r = r + ((v ^ j) * kpx(x, j, qx, w))
Next j

adospuntos_x_m = r - 11 / 24


End Function
'Anualidad temporal que paga al inicio del período (por m-ésimos)
Function adospuntos_xn_m(x As Integer, i As Double, qx As Range, w As Integer, n As Integer)
Dim j As Integer
Dim v As Double
Dim r As Double

j = 0
r = 0
v = (1 + i) ^ (-1)

For j = 0 To n - 1
r = r + ((v ^ j) * kpx(x, j, qx, w))
Next j

adospuntos_xn_m = r - 11 / 24


End Function

'Calcula la anualidad diferida individual pagada por m-ésimos
' n es el parámetro de diferimiento
Function n_adospuntos_x_m(x As Integer, n As Integer, i As Double, qx As Range, w As Integer)
v = (1 + i) ^ (-1)
n_adospuntos_x_m = (adospuntos_x(x + n, i, qx, w) - 11 / 24) * kpx(x, n, qx, w) * v ^ n

End Function


'Calcula la anualidad diferida conjunta aproximada por m-ésimos
'n es el período de diferimiento
Function n_adospuntos_xy_m(x As Integer, y As Integer, n As Integer, i As Double, qx As Range, w As Integer)
Dim j As Integer
Dim v As Double
Dim r As Double
Dim k As Double

j = n
r = 0
v = (1 + i) ^ (-1)

  
For j = n To w - 1
r = r + ((v ^ j) * kpx(x, j, qx, w) * kpx(y, j, qx, w))
Next j

n_adospuntos_xy_m = r - 11 / 24

End Function
